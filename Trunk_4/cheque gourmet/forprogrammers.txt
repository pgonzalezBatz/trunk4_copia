create table devolucion_cheques(
	id		number not null,
	codtra		number not null,
	numero_cheques	number not null,

	fecha		date not null,
	redistribuido	number(1),
	constraint pk_devolucion_cheques primary key (id),
	constraint fk_devolucion_tipos foreign key (tipo) references tipo_cheque(id))
